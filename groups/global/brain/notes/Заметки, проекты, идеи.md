    const TelegramBot = require('node-telegram-bot-api');
    
    const token = 'YOUR_TELEGRAM_BOT_TOKEN';
    const bot = new TelegramBot(token, { polling: true });
    
    const bitcoinApiUrl = 'https://api.coindesk.com/v1/bpi/currentprice.json';

      try {
        const response = await fetch(bitcoinApiUrl);
        const data = await response.json();
        
        const bitcoinPrice = data.bpi.USD.rate;
        const message = `Текущий курс биткоина: $${bitcoinPrice}`;
        
        bot.sendMessage(msg.chat.id, message);
      } catch (error) {
        console.error('Ошибка получения данных о курсе биткоина:', error);
        bot.sendMessage(msg.chat.id, 'Произошла ошибка при получении данных о курсе биткоина.');
      }
    });
